class Solution {
    public String solution(String p) {
        if (p.isEmpty()) {
            return "";
        }
        int countOpen = 0;
        int countClose = 0;
        int idx = 0;
        boolean correct = true;
        while (idx < p.length()) {
            if (p.charAt(idx) == '(') {
                countOpen += 1;
            } else {
                countClose += 1;
            }

            if (countClose > countOpen) {
                correct = false;
            }

            idx += 1;

            if (countOpen == countClose) {
                break;
            }
        }

        String u = p.substring(0, idx);
        String v = p.substring(idx);

        if (correct) {
            return u + solution(v);
        } else {
            StringBuilder sb = new StringBuilder("(");
            sb.append(solution(v));
            sb.append(")");
            for (int i = 1; i < u.length() - 1; i++) {
                char c = p.charAt(i);
                if (c == '(') {
                    sb.append(')');
                } else {
                    sb.append('(');
                }
            }
            return sb.toString();
        }
    }
}

```python
def solution(p):
    if not p:
        return ''

    brackets_count = [0, 0]

    for i in range(len(p)):
        if p[i] == '(':
            brackets_count[0] += 1
        else:
            brackets_count[1] += 1

        if brackets_count[0] == brackets_count[1]:
            break

    u, v = p[:i+1], p[i+1:]

    if u[0] == '(':
        v = solution(v)
        return u + v
    else:
        result = "(" + solution(v) + ")"
        for i in range(1, len(u)-1):
            if u[i] == '(':
                result += ')'
            else:
                result += '('

        return result
```
