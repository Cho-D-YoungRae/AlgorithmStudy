import java.util.*;

import static java.lang.Math.*;

class Solution {
    public int[] solution(int brown, int yellow) {
        int[] answer = new int[2];
        int totalArea = brown + yellow;

        for (int h = 3; h <= ceil(sqrt(totalArea)); h++) {
            if ((totalArea % h) == 0) {
                int w = totalArea / h;
                int yellowArea = (w - 2) * (h - 2);
                if (yellowArea == yellow) {
                    answer[0] = w;
                    answer[1] = h;
                    break;
                }
            }
        }
        return answer;
    }
}
