# 4. 미로 탈출

> 152pg

```python
result = float('inf')

if __name__ == '__main__':

    N, M = map(int, input().split())

    graph = []
    for i in range(N):
        graph.append(list(map(int, list(input()))))



    def dfs(y, x, d):
        if not(0 <= y < N and 0 <= x < M) \
                                or graph[y][x] == 0:
            return

        global result
        graph[y][x] = 0

        if y == N-1 and x == M-1:
            result = min(result, d)
        else:
            dfs(y + 1, x, d + 1)
            dfs(y - 1, x, d + 1)
            dfs(y, x + 1, d + 1)
            dfs(y, x - 1, d + 1)

        graph[y][x] = 1


    dfs(0, 0, 1)


    print(result)
```
